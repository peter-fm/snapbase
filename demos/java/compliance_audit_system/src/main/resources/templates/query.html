Compliance-Focused Queries:
SELECT COUNT(*) as customers_with_pii FROM data WHERE ssn IS NOT NULL
SELECT snapshot_name, AVG(credit_score) as avg_score FROM data GROUP BY snapshot_name
SELECT id, first_name, last_name FROM data WHERE annual_income > 100000 LIMIT 10
SELECT COUNT(DISTINCT id) as unique_customers, snapshot_name FROM data GROUP BY snapshot_name
